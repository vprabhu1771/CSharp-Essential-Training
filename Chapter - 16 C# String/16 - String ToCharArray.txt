using System;
using System.IO;

namespace HelloWorld {

   class Program {

      static void Main(string[] args) {
            
         string data = "Icecream";
         char[] result;

         // copies str to result
         result = data.ToCharArray();


         // prints result
         for (int i = 0; i < result.Length; i++) 
         {
            Console.Write(result[i] + ", ");
         }

      }      
   }
}
